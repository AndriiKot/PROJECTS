

1.
select 
	f.title,
	'rating' as src
from 
	film f 
where 	
	f.rating  = 'G'
union all
select 
	f.title,
	'grant' as src
from 
	film f 
		join film_actor fa 
			on(fa.film_id = f.film_id)
		join actor a 
			on(a.actor_id = fa.actor_id)
where 
	a.last_name  = 'Grant'

select 
	f.title
from 	
	film f 	
where f.rating <> 'G'
intersect
select 
	f.title
from 
	film f 
		join film_actor fa 
			on(fa.film_id = f.film_id)
		join actor a
